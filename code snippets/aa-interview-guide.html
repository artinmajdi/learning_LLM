Implement Basic RAG Pipeline

You may be asked to create a basic RAG pipeline that processes documents, generates embeddings, and retrieves relevant information.

# Sample RAG pipeline implementation
import numpy as np
from sentence_transformers import SentenceTransformer
from sklearn.metrics.pairwise import cosine_similarity

# 1. Document chunking function
def chunk_document(text, chunk_size=200, overlap=50):
    """Split document into overlapping chunks."""
    chunks = []
    for i in range(0, len(text), chunk_size - overlap):
        chunk = text[i:i + chunk_size]
        if len(chunk) >= chunk_size / 2:  # Only keep chunks with sufficient content
            chunks.append(chunk)
    return chunks

# 2. Generate embeddings
model = SentenceTransformer('all-MiniLM-L6-v2')

def get_embeddings(texts):
    """Generate embeddings for a list of texts."""
    return model.encode(texts)

# 3. Simple vector database
class SimpleVectorDB:
    def __init__(self):
        self.documents = []
        self.embeddings = []

    def add_documents(self, documents):
        """Add documents and their embeddings to the database."""
        self.documents.extend(documents)
        new_embeddings = get_embeddings(documents)
        if len(self.embeddings) == 0:
            self.embeddings = new_embeddings
        else:
            self.embeddings = np.vstack([self.embeddings, new_embeddings])

    def search(self, query, top_k=3):
        """Search for most similar documents."""
        query_embedding = get_embeddings([query])[0]
        similarities = cosine_similarity([query_embedding], self.embeddings)[0]
        top_indices = np.argsort(similarities)[-top_k:][::-1]
        return [(self.documents[i], similarities[i]) for i in top_indices]

# 4. Query processing
def process_query(db, query, llm_prompt_template):
    """Process a query using RAG."""
    # Retrieve relevant documents
    retrieved_docs = db.search(query)

    # Format prompt with retrieved context
    context = "\n".join([doc for doc, _ in retrieved_docs])
    prompt = llm_prompt_template.format(query=query, context=context)

    # In a real implementation, you would call your LLM here
    return f"LLM would process: {prompt[:100]}..."

# Azure OpenAI Service integration example
from azure.ai.ml import MLClient
from azure.ai.ml.entities import AmlCompute
from azure.identity import DefaultAzureCredential
from azure.ai.ml.dsl import pipeline
from azure.ai.ml import Input, Output

# 1. Authenticate to Azure
credential = DefaultAzureCredential()
ml_client = MLClient(
    credential=credential,
    subscription_id="your-subscription-id",
    resource_group_name="your-resource-group",
    workspace_name="your-workspace-name"
)

# 2. Define compute
compute_name = "rag-compute"
compute_cluster = AmlCompute(
    name=compute_name,
    size="Standard_DS3_v2",
    min_instances=0,
    max_instances=4,
    idle_time_before_scale_down=120
)
ml_client.begin_create_or_update(compute_cluster).result()

# 3. Define RAG pipeline components
@pipeline
def rag_pipeline(documents_path, query_input):
    # Component for document processing
    document_processor = ml_client.components.get(
        name="document_processor",
        version="1.0.0"
    )
    # Component for embedding generation
    embedding_generator = ml_client.components.get(
        name="embedding_generator",
        version="1.0.0"
    )
    # Component for query processing
    query_processor = ml_client.components.get(
        name="query_processor",
        version="1.0.0"
    )

    # Pipeline definition
    processed_docs = document_processor(
        documents=Input(type="uri_folder", path=documents_path)
    )

    embeddings = embedding_generator(
        documents=processed_docs.outputs.processed_documents
    )

    response = query_processor(
        query=query_input,
        document_embeddings=embeddings.outputs.embeddings
    )

    return {"response": response.outputs.response}

# 4. Submit pipeline job
pipeline_job = rag_pipeline(
    documents_path="azureml://datastores/workspaceblobstore/paths/documents/",
    query_input="What are the best practices for IT system maintenance?"
)

pipeline_job = ml_client.jobs.create_or_update(pipeline_job)
pipeline_job_status = ml_client.jobs.get(pipeline_job.name)

Create Evaluation Framework

This task focuses on building metrics to evaluate RAG system performance, including hallucination detection.

# RAG evaluation framework example
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity
from rouge import Rouge

# 1. Context relevance evaluation
def evaluate_context_relevance(query, retrieved_contexts, model):
    """Evaluate how relevant the retrieved contexts are to the query."""
    query_embedding = model.encode([query])[0]
    context_embeddings = model.encode(retrieved_contexts)

    similarities = cosine_similarity([query_embedding], context_embeddings)[0]
    avg_similarity = np.mean(similarities)

    return {
        "avg_similarity": avg_similarity,
        "max_similarity": np.max(similarities),
        "min_similarity": np.min(similarities)
    }

# 2. Hallucination detection
def detect_hallucinations(response, retrieved_contexts, model):
    """Detect potential hallucinations in the response."""
    # Split response into sentences
    sentences = [s.strip() for s in response.split('.') if s.strip()]

    # Get embeddings
    sentence_embeddings = model.encode(sentences)
    context_embeddings = model.encode(retrieved_contexts)

    # For each sentence, check if it has support in the contexts
    hallucination_scores = []
    for sent_emb in sentence_embeddings:
        # Calculate similarity with each context
        similarities = cosine_similarity([sent_emb], context_embeddings)[0]
        # If max similarity is below threshold, might be a hallucination
        hallucination_scores.append(1.0 - np.max(similarities))

    return {
        "hallucination_score_avg": np.mean(hallucination_scores),
        "high_hallucination_sentences": [
            sentences[i] for i, score in enumerate(hallucination_scores)
            if score > 0.7  # Threshold for potential hallucination
        ]
    }

# 3. Answer correctness (if ground truth available)
def evaluate_answer_correctness(response, ground_truth):
    """Evaluate the correctness of the response against ground truth."""
    rouge = Rouge()
    scores = rouge.get_scores(response, ground_truth)

    return {
        "rouge1_f": scores[0]['rouge-1']['f'],
        "rouge2_f": scores[0]['rouge-2']['f'],
        "rougeL_f": scores[0]['rouge-l']['f']
    }

# 4. Complete evaluation
def evaluate_rag_system(query, response, retrieved_contexts, model, ground_truth=None):
    """Complete evaluation of the RAG system."""
    results = {}

    # Context relevance
    results["context_relevance"] = evaluate_context_relevance(
        query, retrieved_contexts, model
    )

    # Hallucination detection
    results["hallucination"] = detect_hallucinations(
        response, retrieved_contexts, model
    )

    # Answer correctness (if ground truth available)
    if ground_truth:
        results["correctness"] = evaluate_answer_correctness(
            response, ground_truth
        )

    # Overall score (weighted combination)
    results["overall_score"] = 0.4 * (1 - results["hallucination"]["hallucination_score_avg"]) + \
                              0.6 * results["context_relevance"]["avg_similarity"]

    return results
